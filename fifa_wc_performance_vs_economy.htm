headers = list(fifa_df)
correlations = list()
for i in headers[1:]:
    for j in headers[2:]:
        if i is not j:
            correlation = fifa_df[i].corr(fifa_df[j])
            if correlation < 0:
                correlation = 0 - correlation
            correlations.append((correlation, 'correlation between '+ i + ' and '+ j + ' is = ' + str(correlation)))
            fifa_df.plot.scatter(x=i, y=j)
correlations.sort(key=lambda tup: tup[0], reverse=True)
